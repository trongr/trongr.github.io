Insertion sort

void insertion(int a[], size_t n){
    for (size_t i = 1; i < n; i++){
        int key = a[i];
        size_t j = i;
        while (0 < j && key < a[j - 1]){
            a[j] = a[j - 1];
            j--;
        }
        a[j] = key;
    }
}
